# builds it and send it to google's container registry
gcloud builds submit --tag gcr.io/aserverinstance/discordbotbuild

# deploy on a compute instance
# list of machine-types : https://cloud.google.com/compute/docs/general-purpose-machines
gcloud compute instances create-with-container bot-instance-1 --container-image gcr.io/aserverinstance/discordbotbuild --machine-type n2-standard-2

# get list of currently running instances
gcloud compute instances list

# delete an instance that's currently running
gcloud compute instances delete bot-instance-1
